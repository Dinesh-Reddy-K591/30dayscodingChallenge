class Solution {
    public boolean isHappy(int n) {
       HashSet<Integer>map=new HashSet<>();
       while(n!=1 && !map.contains(n)){
        map.add(n);
        n=Sum(n);
       }
        return n==1;
       }
       public int Sum(int n){
        int sum=0;
        while(n>0){
            int digits=n%10;
            sum+=digits*digits;
            n=n/10;
        }
        return sum;
       } 
    }
